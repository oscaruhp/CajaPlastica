CajaPlastica jQuery

Quick start

1.- Include the files 

    <!-- Bootstrap and JQuery -->
    <link rel="stylesheet" href="bootstrap.min.css">
    <script src="jquery.min.js"> </script>
        
    <!-- Plugin -->
    <link rel="stylesheet" href="CajaPlastica.css">
    <script src="CajaPlastica.js"></script>
2.- Create HTML Element for install CajaPlastica 

     <div id="Resultado"></div>
3.- Call the the plugin on a container as your wish (content in var data) 

            var data = [ {
            "url":"http://img.youtube.com/vi/KnVCNMlOPLw/default.jpg", 
            "link": "https://www.youtube.com/user/dimit28",
            "artist":"ColdPlay",
            "title":"Don't Panic"},
            {
            "url":"http://img.youtube.com/vi/rTZIoSmJ2Us/default.jpg", 
            "link": "https://www.youtube.com/user/dimit28",
            "artist":"La Ley",
            "title":"MTV UNPLUGEED"
            },
            {
            "url":"http://img.youtube.com/vi/UMtFjkaJ2VU/default.jpg", 
            "link": "https://www.youtube.com/user/dimit28",
            "title":"Rolling In The Deep",
            "artist":"Adele"
            }];
            
            $('#Resultado').CajaPlastica(data);
4.- All done!
